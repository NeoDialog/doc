Getting started

You will need to include the neoDialog.css and neoDialog.js files in your page. (head tag)

                            <script type="text/javascript" src="js/neoDialog.min.js"></script>

    <link rel="stylesheet" href="css/neoDialog.css">

simple alert :

                            // simple alert
    let alert = new neoDialog();

    alert.setTitle("create account")
            .setText("Your account was created successfully.")
            .setIcon({ id : nd.icon_success })
            .setShowCloseButton(true)
            .setConfirmButton({text : "Ok" , icon : nd.i_done})
            .show();
                            // or for be fast and less code ( nd || neoDialog || NeoDialog )
    nd.fire({
        title : "create account",
        text : "Your account was created successfully.",
        icon : { id : nd.icon_success },
        confirm : {text : "Ok" , icon : nd.i_done},
        closeButton : true
    });

Browsers

This library only works on browsers that support JavaScript ES5 or higher.

nd.fire({ rtl : true });

nd.fire({ closeButton : true });

cancel e.isCancel text , icon

reject e.isReject text , icon 
nd.fire({ reject : { text : "reject" , icon : nd.i_reject } });

Master Icons :

To use the icons, download the Master Icon font. And put the following code in your style file. You can find the files on the Library GitHub page: 
                            @font-face {

  src: url("../master-icons/Master-Icon.eot?2k3mq3");
  src: url("../master-icons/Master-Icon.eot?2k3mq3#iefix") format("embedded-opentype"),
  url("../master-icons/Master-Icon.ttf?2k3mq3") format("truetype"),
  url("../master-icons/Master-Icon.woff?2k3mq3") format("woff"),
  url("../master-icons/Master-Icon.svg?2k3mq3#Master-Icon") format("svg");
  font-weight: normal;
  font-style: normal;
  font-display: block;
}

Example :

const options = {
  title : "Buttons",
  text : "This is an example for buttons and events.",
  closeButton : true,
  confirm : { text : "done" , icon : nd.i_done },
  cancel : { text : "cancel" , icon : nd.i_cancel },
  reject : { text : "reject" , icon : nd.i_reject },
  desc : "You can only use the buttons in dialogs, alerts and snackbars."
};

let alert = new nd(options);

alert.show( e => {
   if (e.isConfirm) {
       nd.toast.success({title : "Buttons" , text : "You clicked the 'confirm' button."});
   } else if (e.isCancel) {
       nd.toast.warning({title : "Buttons" , text : "You clicked the 'cancel' button."});
   } else if (e.isReject) {
       nd.toast.error({title : "Buttons" , text : "You clicked the 'reject' button."});
   }
   console.log(e.values); // if you set input , select & ...
});

const options = {

      id : "pass",
      type : "password" ,
      label : "password" ,
      validation : { min : 6 },
      invalid : "You Must Enter a User Name and Password!"
  }

  ]
};

let login = new nd(options);

login.show( e => {

       nd.toast.success({title : "Buttons" , text : "You clicked the 'confirm' button."});

Custom

NeoDialog gives you the ability to use custom elements in dialogs.

If you use this feature, you will no longer be able to access neo-dialog features. (Such as title, text, inputs , desc and buttons)
key typeof example
title HTMLHeadingElement | HTMLDivElement (Identified as a parent) 
nd.fire({ custom : { title : js_html_element } });
text HTMLParagraphElement | HTMLDivElement (Identified as a parent) 
nd.fire({ custom : { text : js_html_element } });
input HTMLInputElement | HTMLDivElement (Identified as a parent) 
nd.fire({ custom : { input : js_html_element } });
button HTMLButtonElement | HTMLDivElement (Identified as a parent) 
nd.fire({ custom : { button : js_html_element } });
desc HTMLParagraphElement | HTMLDivElement (Identified as a parent) 
nd.fire({ custom : { desc : js_html_element } });

Example :

let customButton = document.createElement("button");
customButton.id = "nd-custom-btn";
customButton.innerText = "upload";
customButton.onclick = () => {
    alert.destroy();
    nd.fire({type : nd.type_toast , title:"Custom",text : "You clicked the custom button!",color : nd.color_success});
};
let customTitle = document.createElement("h2");
customTitle.classList.add("nd-custom-title");
customTitle.innerHTML = "Custom Title";

const options = {
    title : "Custom",
    text : "This is an example for CustomElements.",
    closeButton : true,
    custom : {
        title : customTitle ,
        button : customButton
    },
    desc : "in toast type only can use title & text."
};
let alert = nd.fire(options);

Example for less code and more efficiency:

// Toast
nd.toast.success({title : "Toast", text : "a success toast!", timeout : 15000});
nd.toast.error({title : "Toast", text : "a error toast!", timeout : 15000});
nd.toast.warning({title : "Toast", text : "a warning toast!", timeout : 15000});
nd.toast.info({title : "Toast", text : "a info toast!", timeout : 15000});

// Alert
nd.alert.success({title : "alert", text : "a success alert!"});
nd.alert.error({title : "alert", text : "a error alert!"});
nd.alert.warning({title : "alert", text : "a warning alert!"});
nd.alert.info({title : "alert", text : "a info alert!"});

// snackbar
nd.snackbar.cookies({callback : e => {console.log(e)} });
nd.snackbar.subscribe({callback : e => {console.log(e)} });

// dialog
nd.dialog.login({callback : e => {console.log(e)} });
nd.dialog.validationCode({callback : e => {console.log(e)} });
nd.dialog.sign_up({callback : e => {console.log(e)} });
